# Word文档占位符自动填充指南（通用版）

## 设计原则

本指南采用**"先分析，后填充"**的策略，不依赖任何硬编码的索引或命名，通过内容特征自动定位填充位置。

## 一、通用工具库

### 1.1 基础文本操作

```python
from docx import Document
from docx.oxml.ns import qn
import re

def get_paragraph_text(para):
    """获取段落的完整文本（合并所有runs）"""
    if not para.runs:
        return para.text or ''
    return ''.join([run.text for run in para.runs])

def set_paragraph_text(para, new_text):
    """设置段落的完整文本，保留第一个run的格式"""
    if para.runs:
        para.runs[0].text = new_text
        for run in para.runs[1:]:
            run.text = ''
    else:
        para.add_run(new_text)

def get_cell_text(cell):
    """获取单元格文本"""
    if cell.paragraphs:
        return get_paragraph_text(cell.paragraphs[0])
    return ""

def set_cell_text(cell, text):
    """安全设置单元格文本"""
    if cell.paragraphs:
        first_para = cell.paragraphs[0]
        if first_para.runs:
            first_para.runs[0].text = text
            for run in first_para.runs[1:]:
                run.text = ''
        else:
            first_para.text = text
    else:
        cell.text = text

def get_grid_span(cell):
    """获取单元格跨越的列数"""
    tc = cell._tc
    tcPr = tc.tcPr
    if tcPr is not None:
        gridSpan_elem = tcPr.gridSpan
        if gridSpan_elem is not None:
            return int(gridSpan_elem.val)
    return 1
```

### 1.2 智能查找工具

```python
def find_paragraphs_containing(doc, keywords):
    """
    查找包含任一关键词的段落
    返回: [(段落索引, 段落文本, 匹配的关键词), ...]
    """
    results = []
    for i, para in enumerate(doc.paragraphs):
        text = get_paragraph_text(para)
        for kw in keywords:
            if kw in text:
                results.append((i, text, kw))
                break
    return results

def find_tables_by_header(doc, header_keywords):
    """
    通过表头关键词查找表格
    返回: [(表格索引, 匹配到的表头文本), ...]
    """
    results = []
    for i, table in enumerate(doc.tables):
        if not table.rows:
            continue
        # 检查第一行或前两行作为表头
        for row_idx in range(min(2, len(table.rows))):
            row_text = " ".join([get_cell_text(cell) for cell in table.rows[row_idx].cells])
            for kw in header_keywords:
                if kw in row_text:
                    results.append((i, kw))
                    break
    return results

def find_cell_by_label(table, label_keywords, search_in_first_col=True):
    """
    在表格中查找包含标签的单元格，返回其位置
    label_keywords: 标签的可能写法，如 ['投标人名称', '供应商名称', '单位名称']
    返回: (行索引, 列索引) 或 None
    """
    for ri, row in enumerate(table.rows):
        cells_to_search = [row.cells[0]] if search_in_first_col else row.cells
        for ci, cell in enumerate(cells_to_search):
            text = get_cell_text(cell).strip()
            for kw in label_keywords:
                if kw in text:
                    return (ri, ci)
    return None

def find_next_empty_cell(row, start_idx=0):
    """
    从指定位置开始，找到行中下一个空白单元格
    返回: 单元格对象 或 None
    """
    for i in range(start_idx, len(row.cells)):
        cell = row.cells[i]
        text = get_cell_text(cell).strip()
        if not text or text in ['', '　', ' ']:
            return cell
    return None
```

## 三、通用处理策略

### 3.1 括号型占位符处理（通用）

```python
def fill_bracket_placeholders(doc, data, config=BRACKET_PLACEHOLDERS):
    """
    通用括号型占位符填充
    自动匹配不同文件的命名差异
    """
    fill_count = {}
    
    # 遍历所有段落
    for para in doc.paragraphs:
        text = get_paragraph_text(para)
        for data_key, placeholder_list in config.items():
            if data_key not in data:
                continue
            for placeholder in placeholder_list:
                if placeholder in text:
                    new_text = text.replace(placeholder, data[data_key])
                    set_paragraph_text(para, new_text)
                    fill_count[data_key] = fill_count.get(data_key, 0) + 1
                    break
    
    # 遍历所有表格
    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                text = get_cell_text(cell)
                for data_key, placeholder_list in config.items():
                    if data_key not in data:
                        continue
                    for placeholder in placeholder_list:
                        if placeholder in text:
                            new_text = text.replace(placeholder, data[data_key])
                            set_cell_text(cell, new_text)
                            fill_count[data_key] = fill_count.get(data_key, 0) + 1
                            break
    
    return fill_count
```

### 3.2 下划线日期处理（通用）

```python
def fill_date_placeholders(doc, date_data):
    """
    通用日期占位符填充
    date_data: {'year': '2025', 'month': '01', 'day': '15'}
    """
    replacements = {
        '_年_月_日': f"{date_data['year']}年{date_data['month']}月{date_data['day']}日",
        '_年 _月 _日': f"{date_data['year']}年 {date_data['month']}月 {date_data['day']}日",
        '_年': f"{date_data['year']}年",
        '_月': f"{date_data['month']}月",
        '_日': f"{date_data['day']}日",
    }
    
    # 在段落中替换
    for para in doc.paragraphs:
        text = get_paragraph_text(para)
        new_text = text
        for old, new in replacements.items():
            new_text = new_text.replace(old, new)
        if new_text != text:
            set_paragraph_text(para, new_text)
    
    # 在表格中替换
    for table in doc.tables:
        for row in table.rows:
            for cell in row.cells:
                text = get_cell_text(cell)
                new_text = text
                for old, new in replacements.items():
                    new_text = new_text.replace(old, new)
                if new_text != text:
                    set_cell_text(cell, new_text)
```

### 3.3 标签型占位符处理（通用）

```python
def fill_label_placeholders(doc, data, config=LABEL_PATTERHOLDERS):
    """
    通用标签型占位符填充
    通过标签匹配，支持多种标签写法
    """
    for para in doc.paragraphs:
        text = get_paragraph_text(para).strip()
        
        for data_key, (label_list, data_source) in config.items():
            for label in label_list:
                # 精确匹配标签，或标签后内容很少（说明是空的）
                if text == label or (text.startswith(label) and len(text) <= len(label) + 2):
                    new_value = label + data.get(data_source, '')
                    set_paragraph_text(para, new_value)
                    break
```

### 3.4 表格智能填充（通用）

```python
def fill_table_smart(table, data, field_labels=TABLE_FIELD_LABELS):
    """
    智能表格填充 - 不依赖行列索引
    通过标签匹配自动定位填充位置
    """
    fill_results = {}
    
    for row in table.rows:
        # 遍历行中的每个单元格，查找标签
        for ci, cell in enumerate(row.cells):
            cell_text = get_cell_text(cell).strip()
            
            # 检查是否是标签单元格
            for data_key, label_list in field_labels.items():
                if data_key not in data:
                    continue
                    
                # 匹配标签
                is_match = False
                for label in label_list:
                    if label in cell_text:
                        is_match = True
                        break
                
                if is_match:
                    # 找到标签后的下一个空白单元格
                    # 考虑合并单元格的情况
                    next_idx = ci + 1
                    while next_idx < len(row.cells):
                        target_cell = row.cells[next_idx]
                        target_text = get_cell_text(target_cell).strip()
                        
                        # 如果该单元格是另一个标签，跳过
                        is_another_label = False
                        for other_labels in field_labels.values():
                            if any(l in target_text for l in other_labels):
                                is_another_label = True
                                break
                        
                        if not is_another_label:
                            # 填充数据
                            set_cell_text(target_cell, data[data_key])
                            fill_results[data_key] = True
                            break
                        
                        next_idx += 1
    
    return fill_results

def fill_specific_table(doc, table_keywords, data, field_labels=TABLE_FIELD_LABELS):
    """
    填充特定表格 - 通过表头关键词定位
    """
    # 查找表格
    tables = find_tables_by_header(doc, table_keywords)
    
    if not tables:
        print(f"未找到包含关键词 {table_keywords} 的表格")
        return False
    
    for table_idx, matched_kw in tables:
        print(f"找到表格 {table_idx}，匹配关键词: {matched_kw}")
        table = doc.tables[table_idx]
        results = fill_table_smart(table, data, field_labels)
        print(f"填充结果: {results}")
    
    return True
```

## 四、文档结构分析工具

### 4.1 自动生成文档分析报告

```python
def analyze_document_structure(doc_path, output_report=True):
    """
    分析文档结构，生成配置建议
    """
    doc = Document(doc_path)
    
    report = {
        '段落数': len(doc.paragraphs),
        '表格数': len(doc.tables),
        '发现的括号型占位符': [],
        '发现的下划线日期': [],
        '表格结构': [],
    }
    
    # 1. 分析段落中的占位符
    bracket_pattern = re.compile(r'（[^）]+）')
    for i, para in enumerate(doc.paragraphs):
        text = get_paragraph_text(para)
        matches = bracket_pattern.findall(text)
        for m in matches:
            if m not in report['发现的括号型占位符']:
                report['发现的括号型占位符'].append(m)
    
    # 2. 分析下划线日期
    for i, para in enumerate(doc.paragraphs):
        text = get_paragraph_text(para)
        if '_年' in text or '_月' in text or '_日' in text:
            report['发现的下划线日期'].append(f"段落{i}: {text[:50]}")
    
    # 3. 分析表格结构
    for i, table in enumerate(doc.tables):
        if not table.rows:
            continue
        
        table_info = {
            '索引': i,
            '行数': len(table.rows),
            '网格列数': len(table._tbl.tblGrid.findall(qn('w:gridCol'))),
            '首行内容': [],
        }
        
        # 记录首行或前两行内容
        for ri in range(min(2, len(table.rows))):
            row_content = []
            for ci, cell in enumerate(table.rows[ri].cells):
                text = get_cell_text(cell)[:20]
                span = get_grid_span(cell)
                row_content.append(f"[{ci}]{text}(span={span})")
            table_info['首行内容'].append(row_content)
        
        report['表格结构'].append(table_info)
    
    # 输出报告
    if output_report:
        print("=" * 80)
        print("文档结构分析报告")
        print("=" * 80)
        for key, value in report.items():
            print(f"\n【{key}】")
            if isinstance(value, list):
                for item in value:
                    print(f"  - {item}")
            else:
                print(f"  {value}")
    
    return report
```

## 五、完整处理流程

```python
def process_document(input_path, output_path, data):
    """
    通用文档处理主流程
    """
    print(f"正在处理: {input_path}")
    doc = Document(input_path)
    
    # 步骤1: 分析文档结构（可选，首次处理时运行）
    # analyze_document_structure(doc)
    
    # 步骤2: 填充括号型占位符
    print("\n步骤1: 填充括号型占位符...")
    bracket_results = fill_bracket_placeholders(doc, data)
    print(f"  填充了 {len(bracket_results)} 种占位符")
    
    # 步骤3: 填充日期占位符
    print("\n步骤2: 填充日期占位符...")
    date_data = {
        'year': data.get('投标日期年', '2025'),
        'month': data.get('投标日期月', '01'),
        'day': data.get('投标日期日', '15'),
    }
    fill_date_placeholders(doc, date_data)
    
    # 步骤4: 填充标签型占位符
    print("\n步骤3: 填充标签型占位符...")
    fill_label_placeholders(doc, data)
    
    # 步骤5: 智能填充表格
    print("\n步骤4: 智能填充表格...")
    
    # 尝试填充各种可能的表格
    table_mappings = [
        (TABLE_HEADER_KEYWORDS['投标人基本情况表'], '投标人基本情况表'),
        (TABLE_HEADER_KEYWORDS['法定代表人信息表'], '法定代表人信息表'),
        (TABLE_HEADER_KEYWORDS['项目情况表'], '项目情况表'),
        (TABLE_HEADER_KEYWORDS['项目负责人简历'], '项目负责人简历'),
    ]
    
    for keywords, table_name in table_mappings:
        print(f"\n  查找 {table_name}...")
        fill_specific_table(doc, keywords, data)
    
    # 保存
    print(f"\n保存到: {output_path}")
    doc.save(output_path)
    print("完成!")
    
    return doc
```

### Q1: 某些占位符没有被替换？

检查占位符是否使用不同的写法：
```python
# 查看文档中实际使用的占位符
analyze_document_structure(doc)['发现的括号型占位符']
# 然后添加到 BRACKET_PLACEHOLDERS 对应条目中
```

### Q2: 表格填充位置不对？

先分析表格实际结构：
```python
table = doc.tables[0]  # 或其他索引
for ri, row in enumerate(table.rows):
    for ci, cell in enumerate(row.cells):
        text = get_cell_text(cell)
        span = get_grid_span(cell)
        print(f"Row{ri},Col{ci}: '{text}' span={span}")
```

## 八、总结

本通用指南的核心特点：

1. **零硬编码索引** - 通过内容搜索定位，不依赖表格顺序或行列位置
2. **命名兼容** - 支持多种占位符写法（招标人/采购人/甲方等）
3. **自适应结构** - 自动处理合并单元格，动态确定数据填充位置
4. **可扩展配置** - 通过修改配置映射即可支持新文件格式

使用本指南，只需对新文件运行一次分析，调整配置映射，即可实现自动化填充。
